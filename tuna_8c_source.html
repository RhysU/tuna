93 tuna_stats_mom_result
94 tuna_stats_mom(const tuna_stats* const stats)
95 {
96  tuna_stats_mom_result result;
97  result.n = stats->n;
98  switch (stats->n) {
99  case 0:
100  result.avg = NAN;
101  result.var = NAN;
102  break;
103  case 1:
104  result.avg = stats->m;
105  result.var = 0;
106  break;
107  default:
108  result.avg = stats->m;
109  result.var = stats->s / (stats->n - 1);
110  break;
111  }
112  return result;
113 }

115 void
116 tuna_stats_obs(tuna_stats* const stats,
117  const double x)
118 {
119  /* Algorithm from Knuth TAOCP vol 2, 3rd edition, page 232. */
120  /* Knuth shows better behavior than Welford 1962 on test data. */
121  const size_t n = ++(stats->n);
122  if (n > 1) { /* Second and subsequent invocation */
123  double d = x - stats->m;
124  stats->m += d / n;
125  stats->s += d * (x - stats->m);
126  } else { /* First invocation requires special treatment */
127  stats->m = x;
128  stats->s = 0;
129  }
130 }

143 void
144 tuna_stats_merge(tuna_stats* const dst,
145  const tuna_stats* const src)
146 {
147  if (dst == src) { /* merging with self is a NOP */
148  /* NOP */
149  } else if (src->n == 0) { /* src contains no data */
150  /* NOP */
151  } else if (dst->n == 0) { /* dst contains no data */
152  *dst = *src;
153  } else { /* merge src into dst */
154  size_t total = dst->n + src->n;
155  double dM = dst->m - src->m; /* Cancellation issues? */
156  dst->m = (dst->n * dst->m + src->n * src->m) / total;
157  dst->s = (dst->n == 1 ? 0 : dst->s)
158  + (src->n == 1 ? 0 : src->s)
159  + ((dM * dM) * (dst->n * src->n)) / total;
160  dst->n = total;
161  }
162 }

182 void
183 tuna_chunk_obs(tuna_chunk* const chunk,
184  double cost)
185 {
186  /* First, find smallest observation among set {cost, chunk->outliers[0], ... } */
187  /* placing it into cost while maintaining sorted-ness of chunk->outliers. */
188  /* The loop is one sort pass with possibility of short-circuiting. */
189  if (enforce_lt(&cost, chunk->outliers)) {
190  size_t i;
191  for (i = 1;
192  i < tuna_countof(chunk->outliers)
193  && enforce_lt(chunk->outliers - 1 + i, chunk->outliers + i);
194  ++i)
195  ;
196  }
197 
198  /* Second, when non-zero, record statistics about the best observation. */
199  if (cost) {
200  tuna_stats_obs(&chunk->stats, cost);
201  }
202 
203  /* Together, the above steps cause a zero-initialized tuna_chunk to */
204  /* gather tuna_noutliers pieces of information before beginning to */
205  /* track any statistics. This effectively provides some "start up" */
206  /* or "burn in" period in addition to preventing highly improbable */
207  /* observations from unduly inflating the discovered variability. */
208 }

210 void
211 tuna_chunk_merge(tuna_stats* const stats,
212  const tuna_chunk* const chunk)
213 {
214  size_t i;
215  tuna_stats_merge(stats, &chunk->stats);
216  for (i = 0; i < tuna_countof(chunk->outliers); ++i) {
217  if (chunk->outliers[i]) {
218  tuna_stats_nobs(stats, chunk->outliers + i, tuna_countof(chunk->outliers) - i);
219  break;
220  }
221  }
222 }

243 static
244 double
245 ltqnorm(const double p)
246 {
247  /* Coefficients in rational approximations. */
248  static const double a1 = -3.969683028665376e+01;
249  static const double a2 = 2.209460984245205e+02;
250  static const double a3 = -2.759285104469687e+02;
251  static const double a4 = 1.383577518672690e+02;
252  static const double a5 = -3.066479806614716e+01;
253  static const double a6 = 2.506628277459239e+00;
254 
255  static const double b1 = -5.447609879822406e+01;
256  static const double b2 = 1.615858368580409e+02;
257  static const double b3 = -1.556989798598866e+02;
258  static const double b4 = 6.680131188771972e+01;
259  static const double b5 = -1.328068155288572e+01;
260 
261  static const double c1 = -7.784894002430293e-03;
262  static const double c2 = -3.223964580411365e-01;
263  static const double c3 = -2.400758277161838e+00;
264  static const double c4 = -2.549732539343734e+00;
265  static const double c5 = 4.374664141464968e+00;
266  static const double c6 = 2.938163982698783e+00;
267 
268  static const double d1 = 7.784695709041462e-03;
269  static const double d2 = 3.224671290700398e-01;
270  static const double d3 = 2.445134137142996e+00;
271  static const double d4 = 3.754408661907416e+00;
272 
273  /* Define break-points. */
274  const double p_low = 0.02425;
275  const double p_high = 1 - p_low;
276 
277  /* Compute rational approximation for... */
278  double x, q, r, e, u;
279  if (p < 0) { /* ...domain error. */
280  errno = EDOM;
281  return 0;
282 #ifdef __INTEL_COMPILER
283 #pragma warning(push,disable:1572)
284 #endif
285  } else if (p == 0) { /* ...improbable point. */
286 #ifdef __INTEL_COMPILER
287 #pragma warning(pop)
288 #endif
289  errno = ERANGE;
290 #ifdef HUGE_VAL
291  return -HUGE_VAL;
292 #else
293  return -DBL_MAX;
294 #endif
295  } else if (p < p_low) { /* ...lower region. */
296  q = sqrt(-2 * log(p));
297  x = (((((c1 * q + c2) * q + c3) * q + c4) * q + c5) * q + c6) /
298  ((((d1 * q + d2) * q + d3) * q + d4) * q + 1);
299  } else if (p <= p_high) { /* ...central region */
300  q = p - 0.5;
301  r = q * q;
302  x = (((((a1 * r + a2) * r + a3) * r + a4) * r + a5) * r + a6) * q /
303  (((((b1 * r + b2) * r + b3) * r + b4) * r + b5) * r + 1);
304  } else if (p < 1) { /* ...upper region. */
305  q = sqrt(-2 * log(1 - p));
306  x = -(((((c1 * q + c2) * q + c3) * q + c4) * q + c5) * q + c6) /
307  ((((d1 * q + d2) * q + d3) * q + d4) * q + 1);
308 #ifdef __INTEL_COMPILER
309 #pragma warning(push,disable:1572)
310 #endif
311  } else if (p == 1) { /* ...improbable point. */
312 #ifdef __INTEL_COMPILER
313 #pragma warning(pop)
314 #endif
315  errno = ERANGE;
316 #ifdef HUGE_VAL
317  return HUGE_VAL;
318 #else
319  return DBL_MAX;
320 #endif
321  } else { /* ...domain error. */
322  errno = EDOM;
323  return 0;
324  }
325 
326  /* One hit with Halley's rational method improves to machine precision. */
327  e = 0.5 * erfc(x * -M_SQRT1_2) - p;
328 #define SQRT_2PI (2.50662827463100050240)
329  u = e * SQRT_2PI * exp(x * x / 2);
330 #undef SQRT_2PI
331  x = x - u / (1 + x * u / 2);
332 
333  return x;
334 }

353 static
354 double
355 as3(double t, int nu)
356 {
357  double r, a, b, c, f, s, fk;
358  int i, k, ks, im2, ioe;
359 
360  assert(nu > 0); /* Use errno with EDOM instead? */
361 
362  f = nu;
363  a = t / sqrt(f);
364  r = t;
365  b = f / (f + r * r);
366  im2 = nu - 2;
367  ioe = nu % 2;
368  s = 1;
369  c = 1;
370  f = 1;
371  ks = ioe + 2;
372  fk = ks;
373  if (im2 < 2) {
374  goto L20;
375  }
376  i = im2;
377  for (k = ks; k <= i; k += 2) {
378  c = c * b * (fk - 1) / fk;
379  s += c;
380  if (s == f) {
381  goto L20;
382  }
383  f = s;
384  fk += 2;
385  }
386 L20:
387  if (ioe == 1) {
388  goto L30;
389  }
390  return 0.5 + 0.5 * a * sqrt(b) * s;
391 L30:
392  if (nu == 1) {
393  s = 0;
394  }
395  return 0.5 + (a * b * s + atan(a)) * M_1_PI;
396 }

423 tuna_welch_result
424 tuna_welch(double xA, double sA2, size_t nA,
425  double xB, double sB2, size_t nB)
426 {
427  tuna_welch_result result;
428  double sA2_nA = sA2 / nA;
429  double sB2_nB = sB2 / nB;
430  double t_den2 = sA2_nA + sB2_nB;
431  double nu_den = sA2_nA * sA2_nA / (nA - 1)
432  + sB2_nB * sB2_nB / (nB - 1);
433  result.t = (xA - xB) / sqrt(t_den2);
434  result.nu = t_den2 * t_den2 / nu_den;
435  return result;
436 }

452 double
453 tuna_welch1_approx(double xA, double sA2, size_t nA,

455 {

491 /* Thompson Sampling for Gaussian rewards adapted for minimization.
492  * Samples from posterior Normal(mean, std/sqrt(n)) for each arm,
493  * selecting the arm with lowest sample. */
494 static size_t
495 tuna_algo_thompson_impl(const size_t nchunk,
496  const tuna_chunk* chunks,
497  const double* u01)
498 {
499  size_t i, best;
500  double sample, best_sample, u;
501 
502  assert(nchunk > 0);
503 
504  for (i = 0; i < nchunk; ++i) {
505  if (chunks[i].stats.n < 2) {
506  return i;
507  }
508  }
509 
510  best = 0;
511  for (i = 0; i < nchunk; ++i) {
512  tuna_stats_mom_result s = tuna_stats_mom(&chunks[i].stats);
513  u = u01[i] < 0.001 ? 0.001 : (u01[i] > 0.999 ? 0.999 : u01[i]);
514  sample = s.avg + sqrt(s.var / s.n) * ltqnorm(u);
515  if (i == 0 || sample < best_sample) {
516  best = i;
517  best_sample = sample;
518  }
519  }
520  return best;
521 }

523 /* UCB1 for minimization using Lower Confidence Bound.
524  * LCB(i) = mean(i) - sqrt(2 * ln(total) / n_i) */
525 static size_t
526 tuna_algo_ucb1_impl(const size_t nchunk,
527  const tuna_chunk* chunks,
528  const double* u01)
529 {
530  size_t i, best, total;
531  double lcb, best_lcb, log_total;
532 
533  (void) u01;
534  assert(nchunk > 0);
535 
536  total = 0;
537  for (i = 0; i < nchunk; ++i) {
538  if (chunks[i].stats.n < 2) {
539  return i;
540  }
541  total += chunks[i].stats.n;
542  }
543 
544  log_total = log((double)total);
545  best = 0;
546  for (i = 0; i < nchunk; ++i) {
547  tuna_stats_mom_result s = tuna_stats_mom(&chunks[i].stats);
548  lcb = s.avg - sqrt(2.0 * log_total / s.n);
549  if (i == 0 || lcb < best_lcb) {
550  best = i;
551  best_lcb = lcb;
552  }
553  }
554  return best;
555 }

557 /* Very similar to tuna_algo_welch1_nuinf(...) */
558 static size_t
559 tuna_algo_welch1_nuinf_impl(const size_t nchunk,
560  const tuna_chunk* chunks,
561  const double* u01)
562 {
563  size_t i, j;
564  tuna_stats_mom_result istats, jstats;
565  double p;
566  const size_t ncomparisons = nchunk - 1;
567 
568  assert(nchunk > 0);
569  i = 0;
570  istats = tuna_stats_mom(&chunks[0].stats);
571  if (istats.n < 2) {
572  return i;
573  }
574  for (j = 1; j < nchunk; ++j) {
575  jstats = tuna_stats_mom(&chunks[j].stats);
576  if (jstats.n < 2) {
577  return j;
578  }
579  p = tuna_welch1_nuinf(istats.avg, istats.var, istats.n,
580  jstats.avg, jstats.var, jstats.n);
581  /* Apply Bonferroni correction for multiple comparisons writing */
582  /* p * ncomparisons < u01[j] instead of p < u01[j] / ncomparisons */
583  if (p * ncomparisons < u01[j]) {
584  i = j;
585  istats = jstats;
586  }
587  }
588  return i;
589 }

778 size_t
779 tuna_pre(tuna_site* site,
780  tuna_stack* stack,
781  const tuna_chunk* chunks,
782  const size_t nchunk)
783 {
784  struct timespec ts;
785 
786  /* Delegate chunk selection to tuna_pre_cost */
787  tuna_pre_cost(site, stack, chunks, nchunk);
788 
789  /* Glimpse at the clock so we may compute elapsed time in tuna_post(). */
790  /* Done after algorithm performed to avoid accumulating its runtime. */
791  clock_gettime(TUNA_CLOCK, &ts);
792 
793  /* Stash the time in the opaque stack->ts buffer */
794  /* (after being certain the buffer is adequately sized). */
795  tuna_assert_static(sizeof(stack->ts) >= sizeof(struct tuna_timespec_minimal));
796  memcpy(stack->ts, &ts, sizeof(struct tuna_timespec_minimal));
797 
798  return stack->ik;
799 }

801 double
802 tuna_post(const tuna_stack* stack,
803  tuna_chunk* chunks)
804 {
805  struct timespec ts, te;
806  double elapsed;
807 
808  /* Glimpse at the clock before bookkeeping to ignore its overhead. */
809  clock_gettime(TUNA_CLOCK, &te);
810 
811  /* Retrieve tuna_pre time from the opaque stack->ts buffer */
812  /* (after being certain the buffer is adequately sized). */
813  tuna_assert_static(sizeof(stack->ts) >= sizeof(struct tuna_timespec_minimal));
814  memcpy(&ts, stack->ts, sizeof(struct tuna_timespec_minimal));
815 
816  /* Compute double-valued elapsed time */
817  elapsed = te.tv_nsec - ts.tv_nsec; /* Nanoseconds... */
818  elapsed *= 1e-9; /* ...to seconds */
819  elapsed += te.tv_sec - ts.tv_sec; /* ...plus seconds */
820 
821  /* Delegate recording the observation to tuna_post_cost() */
822  tuna_post_cost(stack, chunks, elapsed);
823 
824  return elapsed;
825 }
